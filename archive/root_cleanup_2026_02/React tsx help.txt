import React, { useState } from 'react';
import { Card, CardContent } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { 
  ImageIcon, Clock, Share2, Grid3x3, Eye, MessageCircle, X, Send, 
  FolderOpen, Users, Mountain, Box, Palette, Camera, Sun, LayoutTemplate,
  ChevronDown, ChevronRight, Search, Plus, Sparkles, Play
} from 'lucide-react';

interface Shot {
  id: number;
  title: string;
  duration: string;
  imagePrompt: string;
  animationPrompt: string;
  thumbnail?: string;
}

interface SequencePlanViewProps {
  sequenceTitle?: string;
  shots?: Shot[];
  onShotClick?: (shotId: number) => void;
  onModify?: () => void;
  onShare?: () => void;
  onExport?: () => void;
  onGenerateSequence?: () => void; // Nouveau callback
}

const SequencePlanView: React.FC<SequencePlanViewProps> = ({
  sequenceTitle = "Plan sequence 1",
  shots = [
    { id: 1, title: "New shot", duration: "5s", imagePrompt: "Prompt text image et animation", animationPrompt: "" },
    { id: 2, title: "New shot", duration: "5s", imagePrompt: "Prompt text image et animation", animationPrompt: "" }
  ],
  onShotClick,
  onModify,
  onShare,
  onExport,
  onGenerateSequence
}) => {
  const [selectedShot, setSelectedShot] = useState<number | null>(null);
  const [isChatOpen, setIsChatOpen] = useState(false);
  const [chatMessage, setChatMessage] = useState('');
  const [chatHistory, setChatHistory] = useState<Array<{role: 'user' | 'assistant', content: string}>>([
    { role: 'assistant', content: 'Bonjour ! Je suis Storycore, votre assistant créatif. Comment puis-je vous aider avec votre projet ?' }
  ]);
  const [expandedSections, setExpandedSections] = useState<string[]>(['assets', 'templates']);

  const handleShotClick = (shotId: number) => {
    setSelectedShot(shotId);
    onShotClick?.(shotId);
  };

  const handleSendMessage = () => {
    if (!chatMessage.trim()) return;
    
    setChatHistory([...chatHistory, { role: 'user', content: chatMessage }]);
    setChatMessage('');
    
    setTimeout(() => {
      setChatHistory(prev => [...prev, { 
        role: 'assistant', 
        content: 'Je traite votre demande...' 
      }]);
    }, 500);
  };

  const toggleSection = (section: string) => {
    setExpandedSections(prev => 
      prev.includes(section) ? prev.filter(s => s !== section) : [...prev, section]
    );
  };

  type TimelineShot = {
    id: number;
    label: string;
    durationSeconds: number;
  };

  const centerTimelineShots: TimelineShot[] = [
    { id: 1, label: 'Shot 1 : Durée 6 secondes', durationSeconds: 6 },
    { id: 2, label: 'Shot 2 : Durée 10 secondes', durationSeconds: 10 },
  ];

  const totalTimelineDuration = centerTimelineShots.reduce(
    (sum, s) => sum + s.durationSeconds,
    0
  );

  return (
    <div className="flex flex-col h-full bg-[#1a1a1a] text-white relative">
      {/* Top Header Bar */}
      <div className="flex items-center justify-between px-4 py-2 bg-[#0f0f0f] border-b border-gray-800">
        <div className="flex items-center gap-4">
          <span className="text-sm font-medium">File</span>
          <span className="text-sm font-medium">Create</span>
          <span className="text-sm font-medium">Edit</span>
          <span className="text-sm font-medium bg-purple-600 px-3 py-1 rounded">View</span>
          <span className="text-sm font-medium">Settings</span>
          <span className="text-sm font-medium">Documentation</span>
          <span className="text-sm font-medium">Help</span>
        </div>
        <div className="flex items-center gap-2">
          <Button variant="outline" size="sm" className="bg-gray-700 border-gray-600">
            <Grid3x3 className="w-4 h-4 mr-1" />
            Partager
          </Button>
          <Button size="sm" className="bg-cyan-500 hover:bg-cyan-600">
            <Share2 className="w-4 h-4 mr-1" />
            Exporter
          </Button>
        </div>
      </div>

      {/* Main Content Area */}
      <div className="flex flex-1 overflow-hidden">
        {/* Left Sidebar */}
        <div className="w-80 bg-[#0f0f0f] border-r border-gray-800 flex flex-col">
          {/* Search Bar */}
          <div className="p-3 border-b border-gray-800">
            <div className="relative">
              <Search className="w-4 h-4 absolute left-3 top-1/2 -translate-y-1/2 text-gray-500" />
              <input 
                type="text" 
                placeholder="Rechercher assets..." 
                className="w-full bg-[#1a1a1a] border border-gray-700 rounded-md py-1.5 pl-9 pr-3 text-xs focus:outline-none focus:border-purple-500"
              />
            </div>
          </div>

          {/* Library Content */}
          <div className="flex-1 overflow-auto custom-scrollbar">
            {/* SECTION ASSETS */}
            <div className="border-b border-gray-800/50">
              <button 
                onClick={() => toggleSection('assets')}
                className="w-full flex items-center justify-between p-3 hover:bg-gray-800/50 transition-colors"
              >
                <div className="flex items-center gap-2">
                  <FolderOpen className="w-4 h-4 text-purple-400" />
                  <span className="text-xs font-bold uppercase tracking-wider text-gray-400">Bibliothèque Assets</span>
                </div>
                {expandedSections.includes('assets') ? <ChevronDown className="w-4 h-4" /> : <ChevronRight className="w-4 h-4" />}
              </button>

              {expandedSections.includes('assets') && (
                <div className="pb-2">
                  <AssetCategory icon={<Users className="w-3.5 h-3.5" />} label="Personnages" count={4} 
                    items={['Protagoniste_A', 'Antagoniste_B', 'PNJ_Foule']} />
                  <AssetCategory icon={<Mountain className="w-3.5 h-3.5" />} label="Environnements" count={12} 
                    items={['Labo_Futuriste', 'Forêt_Brumeuse', 'Cité_Cyberpunk']} />
                  <AssetCategory icon={<Box className="w-3.5 h-3.5" />} label="Props & Objets" count={24} 
                    items={['Artefact_Ancien', 'Véhicule_Solaire']} />
                </div>
              )}
            </div>

            {/* SECTION TEMPLATES */}
            <div className="border-b border-gray-800/50">
              <button 
                onClick={() => toggleSection('templates')}
                className="w-full flex items-center justify-between p-3 hover:bg-gray-800/50 transition-colors"
              >
                <div className="flex items-center gap-2">
                  <LayoutTemplate className="w-4 h-4 text-cyan-400" />
                  <span className="text-xs font-bold uppercase tracking-wider text-gray-400">Templates & Styles</span>
                </div>
                {expandedSections.includes('templates') ? <ChevronDown className="w-4 h-4" /> : <ChevronRight className="w-4 h-4" />}
              </button>

              {expandedSections.includes('templates') && (
                <div className="pb-2">
                  <AssetCategory icon={<Palette className="w-3.5 h-3.5" />} label="Styles Visuels" 
                    items={['Cinématique_Réaliste', 'Noir_et_Blanc', 'Concept_Art']} />
                  <AssetCategory icon={<Camera className="w-3.5 h-3.5" />} label="Presets Caméra" 
                    items={['Travelling_Avant', 'Contre_Plongée', 'Grand_Angle']} />
                  <AssetCategory icon={<Sun className="w-3.5 h-3.5" />} label="Lighting Rig" 
                    items={['Golden_Hour', 'Néon_Nuit', 'Studio_Soft']} />
                </div>
              )}
            </div>
          </div>

          {/* Quick Actions Footer */}
          <div className="p-4 border-t border-gray-800 bg-[#0a0a0a]">
            <Button className="w-full mb-3 bg-purple-600 hover:bg-purple-700 text-xs h-8">
              <Plus className="w-3.5 h-3.5 mr-2" /> Nouvel Asset IA
            </Button>
            <div className="grid grid-cols-2 gap-2">
              <button className="flex flex-col items-center justify-center p-2 bg-gray-800/50 hover:bg-gray-700 rounded border border-gray-700 transition-all">
                <ImageIcon className="w-5 h-5 mb-1 text-cyan-400" />
                <span className="text-[10px]">Dreamina</span>
              </button>
              <button className="flex flex-col items-center justify-center p-2 bg-gray-800/50 hover:bg-gray-700 rounded border border-gray-700 transition-all">
                <MessageCircle className="w-5 h-5 mb-1 text-purple-400" />
                <span className="text-[10px]">Prompt Gen</span>
              </button>
            </div>
          </div>
        </div>

        {/* Center Content */}
        <div className="flex-1 flex flex-col">
          {/* Lecteur Header */}
          <div className="px-6 py-3 border-b border-gray-700 flex items-center justify-between">
            <h2 className="text-lg font-medium">Lecteur</h2>
          </div>

          {/* Video Preview Area */}
          <div className="flex-1 bg-black flex items-center justify-center">
            <div className="text-center text-gray-600">
              <p className="text-sm">Fais glisser les ressources ici et commence à créer</p>
            </div>
          </div>

          {/* Timeline avec visualisation des shots */}
          <div className="px-0 pt-0 pb-3 bg-[#0f0f0f] border-t border-gray-800">
            {/* Barre des commandes */}
            <div className="px-4 py-2 flex items-center gap-2 border-b border-gray-800">
              <button className="p-1 hover:bg-gray-800 rounded text-xs">▶</button>
              <button className="p-1 hover:bg-gray-800 rounded text-xs">⏮</button>
              <button className="p-1 hover:bg-gray-800 rounded text-xs">⏭</button>
              <span className="ml-2 text-xs text-gray-500">Timeline storycore</span>
            </div>

            {/* Bande de timeline avec segments de shots */}
            <div className="px-4 pt-2">
              <div className="relative h-10 w-full bg-[#1c1c1c] border border-gray-700 rounded-sm overflow-hidden flex">
                {centerTimelineShots.map((shot, index) => {
                  const widthPercent = (shot.durationSeconds / totalTimelineDuration) * 100;
                  return (
                    <div
                      key={shot.id}
                      className={`relative h-full bg-purple-600 text-[11px] text-white flex items-center justify-center border-r border-[#c0a3ff] cursor-pointer hover:bg-purple-500 transition-colors ${
                        index === centerTimelineShots.length - 1 ? 'border-r-0' : ''
                      }`}
                      style={{ width: `${widthPercent}%` }}
                      onClick={() => {
                        setSelectedShot(shot.id);
                        onShotClick?.(shot.id);
                      }}
                    >
                      <span className="px-2 truncate">{shot.label}</span>
                    </div>
                  );
                })}

                {/* Bouton + pour ajouter un shot */}
                <button
                  className="h-full w-10 bg-[#c0a3ff] flex items-center justify-center text-black text-xl font-bold hover:bg-[#e0c8ff] transition-colors"
                  onClick={() => console.log('Ajouter un shot')}
                >
                  +
                </button>
              </div>
            </div>

            {/* Légende sous la timeline */}
            <div className="px-4 pt-2">
              <p className="text-xs text-gray-500">
                Fais glisser les ressources ici et commence à créer
              </p>
            </div>
          </div>
        </div>

        {/* Right Panel - Sequence Plans */}
        <div className="w-[600px] bg-[#1a1a1a] border-l border-gray-800 flex flex-col">
          {/* Header avec bouton de génération */}
          <div className="px-4 py-3 border-b border-gray-700 flex items-center justify-between">
            <h2 className="text-base font-medium">{sequenceTitle}</h2>
            <Button 
              onClick={onGenerateSequence}
              size="sm"
              className="bg-gradient-to-r from-purple-600 to-pink-600 hover:from-purple-700 hover:to-pink-700 text-white shadow-lg"
            >
              <Sparkles className="w-4 h-4 mr-2" />
              Générer Séquence
            </Button>
          </div>

          {/* Shots Grid */}
          <div className="flex-1 p-4 overflow-auto">
            <div className="grid grid-cols-2 gap-3">
              {shots.map((shot) => (
                <Card
                  key={shot.id}
                  className={`bg-[#2a2a2a] border-2 cursor-pointer transition-all hover:border-purple-400 ${
                    selectedShot === shot.id ? 'border-purple-500' : 'border-purple-600'
                  }`}
                  onClick={() => handleShotClick(shot.id)}
                >
                  <CardContent className="p-0">
                    {/* Shot Number */}
                    <div className="px-3 py-1.5 border-b border-gray-700">
                      <span className="text-xl font-bold text-gray-300">{shot.id}</span>
                    </div>

                    {/* Thumbnail Area */}
                    <div className="relative aspect-video bg-[#1a1a1a] flex items-center justify-center border-b border-gray-700">
                      {shot.thumbnail ? (
                        <img src={shot.thumbnail} alt={shot.title} className="w-full h-full object-cover" />
                      ) : (
                        <ImageIcon className="w-12 h-12 text-gray-600" />
                      )}
                    </div>

                    {/* Shot Info */}
                    <div className="p-3 space-y-1.5">
                      <div className="flex items-center justify-between text-xs text-gray-400">
                        <span>{shot.title}</span>
                        <div className="flex items-center gap-1">
                          <span>Durée: {shot.duration}</span>
                        </div>
                      </div>
                      <div className="text-xs text-gray-300">
                        {shot.imagePrompt}
                      </div>
                    </div>
                  </CardContent>
                </Card>
              ))}
            </div>
          </div>

          {/* Timeline Preview */}
          <div className="px-4 py-2 border-t border-gray-700">
            <div className="flex items-center gap-2">
              <div className="flex-1 h-1.5 bg-gray-700 rounded-full overflow-hidden">
                <div className="h-full bg-purple-500" style={{ width: '30%' }} />
              </div>
              <span className="text-xs text-gray-400">01:20</span>
            </div>
          </div>

          {/* Footer Info */}
          <div className="px-4 py-3 bg-[#0f0f0f] border-t border-gray-700">
            <div className="grid grid-cols-2 gap-x-4 gap-y-2 text-xs mb-3">
              <div className="flex">
                <span className="text-gray-500 w-32">Paramètres:</span>
                <span className="text-gray-300 truncate">C:\Users\...\Projects\</span>
              </div>
              <div className="flex">
                <span className="text-gray-500 w-32">Format:</span>
                <span className="text-gray-300">Originale</span>
              </div>
              <div className="flex">
                <span className="text-gray-500 w-32">Résolution:</span>
                <span className="text-gray-300">Adaptable</span>
              </div>
              <div className="flex">
                <span className="text-gray-500 w-32">Fréquence d'images:</span>
                <span className="text-gray-300">30.00fps</span>
              </div>
            </div>
            <div className="flex justify-end">
              <Button 
                onClick={onModify}
                size="sm"
                className="bg-purple-600 hover:bg-purple-700 text-white"
              >
                Modifier
              </Button>
            </div>
          </div>
        </div>
      </div>

      {/* Floating Chat Button */}
      <button
        onClick={() => setIsChatOpen(!isChatOpen)}
        className="fixed bottom-6 right-6 w-14 h-14 bg-gradient-to-br from-purple-600 to-cyan-500 rounded-full shadow-lg hover:shadow-xl transition-all hover:scale-110 flex items-center justify-center z-50"
      >
        {isChatOpen ? (
          <X className="w-6 h-6 text-white" />
        ) : (
          <MessageCircle className="w-6 h-6 text-white" />
        )}
      </button>

      {/* Chat Box */}
      {isChatOpen && (
        <div className="fixed bottom-24 right-6 w-96 h-[500px] bg-[#1a1a1a] border-2 border-purple-500 rounded-lg shadow-2xl flex flex-col z-50">
          {/* Chat Header */}
          <div className="px-4 py-3 bg-gradient-to-r from-purple-600 to-cyan-500 rounded-t-lg flex items-center justify-between">
            <div className="flex items-center gap-2">
              <div className="w-8 h-8 bg-white rounded-full flex items-center justify-center">
                <span className="text-purple-600 font-bold text-sm">SC</span>
              </div>
              <div>
                <h3 className="font-semibold text-white">Storycore Assistant</h3>
                <p className="text-xs text-purple-100">En ligne</p>
              </div>
            </div>
            <button 
              onClick={() => setIsChatOpen(false)}
              className="text-white hover:bg-white/20 rounded p-1"
            >
              <X className="w-5 h-5" />
            </button>
          </div>

          {/* Chat Messages */}
          <div className="flex-1 overflow-auto p-4 space-y-3">
            {chatHistory.map((msg, idx) => (
              <div
                key={idx}
                className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}
              >
                <div
                  className={`max-w-[80%] px-4 py-2 rounded-lg ${
                    msg.role === 'user'
                      ? 'bg-purple-600 text-white'
                      : 'bg-gray-800 text-gray-100'
                  }`}
                >
                  <p className="text-sm">{msg.content}</p>
                </div>
              </div>
            ))}
          </div>

          {/* Chat Input */}
          <div className="p-4 border-t border-gray-700">
            <div className="flex gap-2">
              <input
                type="text"
                value={chatMessage}
                onChange={(e) => setChatMessage(e.target.value)}
                onKeyPress={(e) => e.key === 'Enter' && handleSendMessage()}
                placeholder="Posez votre question..."
                className="flex-1 px-3 py-2 bg-gray-800 border border-gray-700 rounded-lg text-sm text-white placeholder-gray-500 focus:outline-none focus:border-purple-500"
              />
              <button
                onClick={handleSendMessage}
                className="px-4 py-2 bg-purple-600 hover:bg-purple-700 rounded-lg transition-colors"
              >
                <Send className="w-4 h-4 text-white" />
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

// Sous-composant pour les catégories
const AssetCategory = ({ icon, label, count, items }: { icon: any, label: string, count?: number, items: string[] }) => {
  const [isOpen, setIsOpen] = useState(false);
  return (
    <div className="px-2">
      <button 
        onClick={() => setIsOpen(!isOpen)}
        className="w-full flex items-center justify-between px-3 py-1.5 hover:bg-gray-800 rounded text-sm text-gray-300 transition-colors"
      >
        <div className="flex items-center gap-2">
          {icon}
          <span className="text-xs">{label}</span>
          {count && <span className="text-[10px] bg-gray-700 px-1.5 rounded-full text-gray-400">{count}</span>}
        </div>
        {isOpen ? <ChevronDown className="w-3 h-3" /> : <ChevronRight className="w-3 h-3" />}
      </button>
      {isOpen && (
        <div className="ml-7 mt-1 space-y-1 border-l border-gray-800">
          {items.map((item, i) => (
            <div key={i} className="px-3 py-1 text-[11px] text-gray-500 hover:text-cyan-400 cursor-pointer truncate">
              • {item}
            </div>
          ))}
        </div>
      )}
    </div>
  );
};

export default SequencePlanView;
